\date{\today}

\begin{abstract}
We study the \emph{$k$-noncorrelation} of automatic sequences arising
from binary pattern counting.  For a finite binary pattern set
$A \subset \{0,1\}^*$ and a root of unity $\omega_k = e^{2\pi i/k}$,
define the 2-automatic sequence $a_{A,k}(n) = \omega_k^{\occ(A,n)}$,
where $\occ(A,n)$ counts occurrences of patterns in $A$ in the binary
expansion of $n$.  The sequence is \emph{$k$-noncorrelated} if all
non-trivial complex self-correlation coefficients vanish.

The paper has two components.  First, we establish exact identities
that connect $k$-noncorrelation to residue-class distributions of
$D_m(n):=\occ(A,n)-\occ(A,n+m)$.  In particular, for $k=4$ we express
the complex correlation in terms of two imbalance parameters
(``mod-$4$ deficits''), which yields transparent necessary and
sufficient conditions at each lag.

Second, we present an exhaustive computational census over 846 pattern
sets (all non-empty subsets of size at most 4 built from patterns of
length at most 4).  In this search space we find 15 sequences that are
$4$-noncorrelated, none that are simultaneously $2$- and $4$-noncorrelated,
and two sequences that are simultaneously $2$- and $6$-noncorrelated.
We also observe no examples for $k\in\{3,5,7,8\}$.

These results support a sharp $2$-NC/$4$-NC dichotomy and indicate that
$k$-noncorrelation is best viewed as a family of distinct spectral
constraints indexed by $k$, rather than as a monotone strengthening
with increasing $k$.
\end{abstract}

\section{Introduction}
\label{sec:intro}
% ══════════════════════════════════════════════════════════════════════════════

\subsection{Background}

A sequence $a:\N_0 \to \C$ is \emph{noncorrelated} if
\[
  \gamma_a(m)
  = \lim_{N \to \infty} \frac{1}{N}\sum_{n=0}^{N-1} a(n)\overline{a(n+m)}
  = 0 \qquad \text{for all } m \geq 1.
\]
For $\{\pm1\}$-valued automatic sequences (sequences computed by a
finite-state machine reading binary digit expansions) this is equivalent
to the spectral measure $\mu_a$ being Lebesgue-equivalent on the circle
$\T$.  The paradigmatic examples are:
\begin{itemize}
  \item \textbf{Rudin--Shapiro} $r(n) = (-1)^{\occ(11, n)}$:
        noncorrelated (flat spectrum).
  \item \textbf{Thue--Morse} $t(n) = (-1)^{s_2(n)}$:
        not noncorrelated ($\limsup_m |\gamma_t(m)| = \tfrac13$).
\end{itemize}
The binary pattern classification problem---given $A$, decide whether
$(-1)^{\occ(A,n)}$ is noncorrelated---was resolved algorithmically by
Konieczny~\cite{Kon21}.

\subsection{The complex-rooted generalisation}

Motivated by the question of whether noncorrelation is a single
phenomenon or a family of independent spectral conditions, we study
the \emph{$k$-rooted} generalisation:
\[
  a_{A,k}(n) = \omega_k^{\occ(A,n)}, \quad \omega_k = e^{2\pi i/k}.
\]
For $k = 2$, this reduces to the standard $\{\pm1\}$-valued sequence.
For $k \geq 3$, $a_{A,k}$ is valued in the $k$-th roots of unity
and remains $2$-automatic, since $n \mapsto \occ(A,n) \bmod k$ is
computed by a finite automaton.

The condition $k$-NC is that $\gamma_{a_{A,k}}(m) = 0$ for all $m \geq 1$,
where
\[
  \gamma_{a_{A,k}}(m)
  = \lim_{N \to \infty} \frac{1}{N} \sum_{n=0}^{N-1}
    \omega_k^{\occ(A,n) - \occ(A,n+m)}.
\]
Writing $D_m(n) = \occ(A,n) - \occ(A,n+m)$, this becomes
$\widehat{f_m}(1) = 0$ where $f_m$ is the limiting distribution of
$D_m(n) \bmod k$ and $\hat{f}_m(j) = \sum_r f_m(r)\omega_k^{jr}$ is
its discrete Fourier transform on $\Z/k\Z$.

\subsection{Overview of results}

\begin{enumerate}[label=(\roman*)]
  \item \textbf{Exact mod-$4$ decomposition} (\S\ref{sec:parity}):
        for each lag we write $\gamma_{a_{A,4}}(m)$ explicitly in terms
        of two residue-class imbalances.  This gives a clean criterion
        for $4$-NC at fixed lag.
  \item \textbf{15 new $k=4$-NC sequences} (\S\ref{sec:k4nc}):
        a complete census of pattern sets of total length at most~$4$,
        yielding four parity classes with distinct algebraic mechanisms.
  \item \textbf{2-NC / 4-NC dichotomy} (\S\ref{sec:dichotomy}):
        no binary pattern sequence is simultaneously $2$-NC and $4$-NC
        in our full search space (846 sets).
  \item \textbf{2+6-NC coexistence} (\S\ref{sec:26nc}):
        two sequences are simultaneously $2$-NC and $6$-NC.  This
        suggests that the dichotomy is specific to $(k=2,k=4)$ rather
        than a universal incompatibility among all even moduli.
  \item \textbf{$k$-NC landscape} (\S\ref{sec:landscape}):
        among $k \in \{2,\ldots,12\}$, only $k \in \{2, 4, 6\}$
        produce examples in our census.
\end{enumerate}

\section{Notation and setup}
\label{sec:notation}
% ══════════════════════════════════════════════════════════════════════════════

\subsection{Pattern sequences}

Let $A \subset \{0,1\}^* = \bigcup_{\ell \geq 1}\{0,1\}^\ell$ be a
finite set of binary \emph{patterns}.  For $n \geq 0$ with binary
expansion $\mathtt{bin}(n)$, let $\occ(A, n)$ denote the total number
of overlapping occurrences of patterns in $A$ in the string
$\mathtt{0}^{L-1}\mathtt{bin}(n)$, where $L = \max_{p \in A}|p|$
(leading-zero padding convention of~\cite{Kon21}).

The \emph{$k$-rooted pattern sequence} is $a_{A,k}(n) = \omega_k^{\occ(A,n)}$.
The \emph{difference process} at lag $m$ is
\[
  D_m(n) = \occ(A,n) - \occ(A,n+m) \in \Z.
\]

\subsection{$k$-noncorrelation}

\begin{definition}[$k$-NC]
The sequence $a_{A,k}$ is \emph{$k$-noncorrelated} ($k$-NC) if
\[
  \gamma_{a_{A,k}}(m)
  \;:=\; \lim_{N \to \infty} \frac{1}{N} \sum_{n=0}^{N-1}
         \omega_k^{D_m(n)} = 0
  \qquad \text{for all } m \geq 1.
\]
The sequence is \emph{not} $k$-NC if this limit is non-zero for some $m$.
\end{definition}

For $k = 2$, $\omega_2 = -1$ and the condition is the classical binary
noncorrelation.  For $k \geq 3$, the limit $\gamma_{a_{A,k}}(m) \in \C$
may be complex-valued.

\begin{remark}
$k$-NC is equivalent to the spectral measure of $a_{A,k}$ being
absolutely continuous on $\T$.  It \emph{does not} in general imply
or is implied by $2$-NC; the relationship between these conditions is
the main object of study.
\end{remark}

\subsection{Empirical estimator}

For sample size $N$ and lag $1 \leq m \leq M$:
\[
  \hat\gamma_{a_{A,k},N}(m)
  = \frac{1}{N}\sum_{n=0}^{N-1} \omega_k^{D_m(n)},
  \qquad
  G_{A,k,N} := \max_{1 \leq m \leq M} |\hat\gamma_{a_{A,k},N}(m)|.
\]
All computations use $M = 64$ lags.  We compute $\hat\gamma$ via
zero-padded FFT, using the full complex modulus $|\cdot|$.  We declare
a sequence \emph{empirically $k$-NC at scale $N$} if $G_{A,k,N} < 0.007$.

% ══════════════════════════════════════════════════════════════════════════════
\section{Mod-$4$ identities and parity taxonomy}
\label{sec:parity}
% ══════════════════════════════════════════════════════════════════════════════

\subsection{What is proved and what is empirical}

In this section we separate exact identities (proved) from structural
patterns seen in the data (empirical).  This distinction is essential:
for general $A$ the parity of $D_m(n)$ is \emph{not} constant in $n$
(e.g. for $A=\{11\}$ one observes both parities already at $m=1$).

\subsection{An exact decomposition for $k=4$}

\begin{proposition}[Lag-wise decomposition at $k=4$]
\label{prop:deficit}
Fix $A$, $m$, and $N$.  Let
\[
  p_r := \Prob_{n\sim U\{0,\dots,N-1\}}\!\bigl[D_m(n)\equiv r \!\!\pmod 4\bigr],
  \qquad r\in\{0,1,2,3\}.
\]
Then
\[
  \hat\gamma_{a_{A,4},N}(m)
  = (p_0-p_2) + i\,(p_1-p_3).
\]
Hence $|\hat\gamma_{a_{A,4},N}(m)|=0$ iff
\[
  p_0=p_2 \quad\text{and}\quad p_1=p_3.
\]
\end{proposition}

\begin{proof}
By definition,
\[
  \hat\gamma_{a_{A,4},N}(m)
  = \frac1N\sum_{n=0}^{N-1} i^{D_m(n)}
  = \sum_{r=0}^3 p_r\, i^r
  = (p_0-p_2) + i(p_1-p_3).
\]
The claimed equivalence follows immediately.
\end{proof}

\begin{definition}[mod-$4$ deficits]
For fixed $(A,m,N)$ define
\[
  \delta_m^{\mathrm{Re}} := p_0-p_2,
  \qquad
  \delta_m^{\mathrm{Im}} := p_1-p_3.
\]
Then $|\hat\gamma_{a_{A,4},N}(m)| = \sqrt{(\delta_m^{\mathrm{Re}})^2+(\delta_m^{\mathrm{Im}})^2}$.
\end{definition}

\begin{corollary}[Constant-parity obstruction at $k=2$]
\label{cor:parity_obstruction}
If for some lag $m$ one has $D_m(n)\equiv p\pmod 2$ for all $n$, then
\[
  \hat\gamma_{a_{A,2},N}(m)=(-1)^p
\]
for every $N$, so $a_{A,2}$ cannot be $2$-noncorrelated.
\end{corollary}

\begin{proof}
Immediate from
$\hat\gamma_{a_{A,2},N}(m)=\frac1N\sum_n(-1)^{D_m(n)}$.
\end{proof}

\subsection{Empirical parity taxonomy for the $k=4$ family}

For the 15 empirically $4$-NC sequences identified in
\S\ref{sec:k4nc}, the lag-wise parity behavior of $D_m(n)$
falls into four recurring classes:
\begin{itemize}
  \item ALL-EVEN;
  \item ALTERNATING parity by lag parity;
  \item MIXED at all tested lags;
  \item one hybrid (MOME) pattern.
\end{itemize}
This taxonomy is an observed regularity in our census, not a theorem
for all binary pattern sets.

\section{The complete $k=4$-NC catalogue}
\label{sec:k4nc}
% ══════════════════════════════════════════════════════════════════════════════

\subsection{Search methodology}

We enumerate all non-empty subsets of $\{w \in \{0,1\}^+ : 1 \leq |w| \leq 3,
1 \in w\}$ of size $1$, $2$, $3$, and $4$.  For each subset $A$ and
each $k \in \{2,3,4,5,6,8\}$ we compute $G_{A,k,N}$ at $N = 2^{12}$
(pre-screen) and verify survivors at $N = 2^{15}$ and $N = 2^{16}$.

\subsection{Four parity classes of $k=4$-NC sequences}

The $15$ confirmed $k=4$-NC sequences divide into four classes based
on their parity signature $\sigma_A = (p_1(A), p_2(A), \ldots)$:

\begin{description}[leftmargin=2cm, labelwidth=2cm]

\item[\textbf{ALL-EVEN.}]
$p_m(A) = 0$ for all $m \geq 1$.  Equivalently, $D_m(n)$ is always
even.  The $k=2$ correlation equals $1$ identically ($2$-NC fails
completely).  The $4$-NC condition reduces to: $D_m \bmod 4$ is
balanced between $\{0, 2\}$ for every $m$.
Three sequences:
$\{01,001,101\}$, $\{10,010,110\}$, $\{11,011,111\}$.

\item[\textbf{ALTERNATING.}]
$p_m(A) = m \bmod 2$ (odd parity at odd lags, even parity at even
lags).  The $k=2$ correlation equals $(-1)^m$ (maximal and alternating).
Four sequences: $\{01,10\}$, $\{01,010,110\}$,
$\{10,001,101\}$, $\{001,010,101,110\}$.

\item[\textbf{MIXED$+$TM.}]
$D_m$ has mixed parity for all $m$ (a necessary condition for $2$-NC),
yet the sequence is still $4$-NC.  The $k=2$ correlation is at the
\emph{Thue-Morse level} $\approx 0.336$, not zero---so these sequences
are not $2$-NC despite having mixed parity.  This shows that mixed
parity is necessary but not sufficient for $2$-NC.  Seven sequences:
$\{1,01,10\}$, $\{1,01,001,101\}$, $\{1,01,010,110\}$,
$\{1,10,001,101\}$, $\{1,10,010,110\}$, $\{1,11,011,111\}$,
$\{001,010,100,111\}$.

\item[\textbf{MOME.}]
Alternating mixed/constant parity (Mixed-Odd-Mixed-Even pattern).
One sequence: $\{001,011,100,110\}$, with parity signature
$(\mathtt{M},1,\mathtt{M},0,\mathtt{M},1,\ldots)$.

\end{description}

\subsection{Minimal example: the transition-parity sequence}

The simplest $k=4$-NC sequence is $a_{\{01,10\},4}(n) = i^{\tau(n)}$,
where $\tau(n) = \occ(\{01,10\}, n)$ counts \emph{binary transitions}
(adjacent digit pairs that differ) in the padded binary expansion of $n$.

\begin{example}
For $n = 13 = (1101)_2$, the padded string is $01101$, with transitions
at positions $(0,1), (2,3), (3,4)$, giving $\tau(13) = 3$ and
$a_{\{01,10\},4}(13) = i^3 = -i$.
\end{example}

This sequence is 2-automatic (since $\tau(n) \bmod 4$ is computed by
a 4-state automaton) and lies in the ALTERNATING class: $D_m(n)$ is
always odd for odd $m$ and always even for even $m$.  The balanced
split within each parity class follows from the equidistribution of
the Gray-code increment structure.

\section{The 2-NC / 4-NC dichotomy}
\label{sec:dichotomy}
% ══════════════════════════════════════════════════════════════════════════════

\subsection{Statement and evidence}

\begin{conjecture}[2-NC / 4-NC dichotomy]
\label{conj:dichotomy}
No binary pattern sequence $a_A$ is simultaneously $2$-noncorrelated
and $4$-noncorrelated.
\end{conjecture}

\textbf{Computational evidence.}
We tested all $846$ pattern sets described in \S\ref{sec:k4nc} (all
subsets of size $1$--$4$ from patterns of length $\leq 4$) for
simultaneous $2$-NC and $4$-NC with threshold $0.007$ at $N = 2^{12}$.
\emph{No violations were found.}  Among the $139$ sequences classified
as $2$-NC, every one has $G_{A,4,N} \approx 0.465$--$0.500$, far from
zero.  Among the $15$ $k=4$-NC sequences, every one has
$G_{A,2,N} \geq 0.334$, far from zero.

\subsection{Mechanism: the mod-4 deficit}

\begin{definition}[4-NC deficit]
For a pattern set $A$, lag $m$, and sample size $N$, the
\emph{$4$-NC deficit} at lag $m$ is
\[
  \delta^{\rm Re}_m(A)
  = \Prob_{n \sim U[N]}[D_m(n) \equiv 0 \pmod 4]
  - \Prob_{n \sim U[N]}[D_m(n) \equiv 2 \pmod 4],
\]
\[
  \delta^{\rm Im}_m(A)
  = \Prob_{n \sim U[N]}[D_m(n) \equiv 1 \pmod 4]
  - \Prob_{n \sim U[N]}[D_m(n) \equiv 3 \pmod 4].
\]
$4$-NC at lag $m$ is equivalent to $\delta^{\rm Re}_m = \delta^{\rm Im}_m = 0$.
\end{definition}

\begin{figure}[ht]
\centering
\includegraphics[width=\textwidth]{dichotomy_mod4_dist.png}
\caption{Mod-$4$ distribution of $D_m(n)$ for $m \in \{1,2,3,4\}$,
$N = 4096$.  \emph{Top row}: four $2$-NC sequences.  The distribution
is highly non-uniform mod $4$; the $4$-NC deficit is large.
\emph{Bottom row}: four $4$-NC sequences.  The distribution concentrates
on $\{p, p+2\}$ with equal weight; the deficit is~$0$.}
\label{fig:mod4dist}
\end{figure}

Figure~\ref{fig:mod4dist} reveals the precise obstruction.  For
$2$-NC sequences:
\begin{itemize}
  \item \textbf{Rudin--Shapiro} $\{11\}$, $m = 1$:
    $P(D \equiv 0) = 0.450$, $P(D \equiv 2) = 0.050$,
    deficit $= 0.400$.
  \item $\{01\}$, $m = 1$: $P(D \equiv 0) = 0.500$,
    $P(D \equiv 2) = 0.000$, deficit $= 0.500$.
  \item $A_3 = \{101,111\}$, $m = 1$: deficit $= 0.425$.
\end{itemize}
These large deficits persist across all lags, so $\sum_n i^{D_m(n)}$
is always far from zero.

For $4$-NC sequences, both components of the deficit are numerically
indistinguishable from~$0$ (within finite-sample fluctuation), which is
consistent with exact mod-$4$ balance.

\subsection{Why incompatibility is plausible}

The decomposition in \Cref{prop:deficit} shows that $4$-NC requires
simultaneous cancellation of two independent imbalances
($\delta_m^{\mathrm{Re}}$ and $\delta_m^{\mathrm{Im}}$) at every lag.
By contrast, $2$-NC requires cancellation of only one parity-level
imbalance at every lag.

Empirically, among all 139 sequences classified as $2$-NC in our
census, both mod-$4$ deficits remain far from zero for at least one
small lag (typically $m=1,2,3$), while all 15 sequences classified as
$4$-NC have large non-zero $k=2$ correlation (Table~\ref{tab:k4full}).
This is the main evidence behind \Cref{conj:dichotomy}.

\begin{remark}
At present we do not have a proof of \Cref{conj:dichotomy}.  The open
problem is to show, in the $2$-regular module generated by
$n\mapsto \occ(A,n)$, that the linear constraints defining $2$-NC and
$4$-NC cannot hold simultaneously for all lags.
\end{remark}

\section{Simultaneously 2-NC and 6-NC: the subgroup mechanism}
\label{sec:26nc}
% ══════════════════════════════════════════════════════════════════════════════

\subsection{The two dual sequences}

Extended search over size-$4$ subsets yields two pattern sets that
are simultaneously $2$-NC and $6$-NC:
\[
  A_1 = \{01, 10, 001, 101\},
  \qquad
  A_2 = \{01, 10, 010, 110\}.
\]
Both are verified at $N = 2^{16}$ with $G_{A,2,2^{16}} = 2.9 \times 10^{-4}$
and $G_{A,6,2^{16}} = 2.3 \times 10^{-4}$, halving ratio $2.00$ in both cases.

\emph{Right}: convergence at $k = 2, 4, 6$, confirming $O(1/N)$
for the NC values and constant for $k=4$.}
\label{fig:dual_nc}
\end{figure}

The $k$-spectrum of $A_1$ (Figure~\ref{fig:dual_nc}) is remarkable:
\begin{align*}
  G_{A_1, 2, N}  &\sim C_2 / N \qquad (k = 2\text{-NC}), \\
  G_{A_1, 3, N}  &\to 1.000     \qquad (\text{maximally anti-flat!}),\\
  G_{A_1, 4, N}  &\to 0.500     \qquad (\text{numerically $\tfrac{1}{2}$}),\\
  G_{A_1, 5, N}  &\to 0.250     \qquad (\text{numerically $\tfrac{1}{4}$}),\\
  G_{A_1, 6, N}  &\sim C_6 / N \qquad (k = 6\text{-NC}),\\
  G_{A_1, 8, N}  &\to 0.354 = \tfrac{1}{\sqrt{8}}
                   \qquad (\text{numerically $2^{-3/2}$}).
\end{align*}
These exact rational and algebraic values strongly suggest that the
non-NC correlations are determined by a precise algebraic structure,
not just asymptotics.

\subsection{Subgroup mechanism (empirical law)}

\begin{proposition}[Observed mod-$6$ law for $A_1$ and $A_2$]
\label{prop:26nc_obs}
For $A_1=\{01,10,001,101\}$ and
$A_2=\{01,10,010,110\}$, computations up to $N=2^{16}$ and
lags $1\le m\le 64$ indicate:
\begin{itemize}
  \item for even $m$, $D_m(n)\bmod 6$ is concentrated near
        $\{0,3\}$ with approximately equal masses;
  \item for odd $m$, $D_m(n)\bmod 6$ is concentrated near
        $\{1,2,4,5\}$ with approximately equal masses.
\end{itemize}
Under this law, both character sums
$\sum_r p_r\omega_6^r$ and $\sum_r p_r(-1)^r$ vanish, explaining the
simultaneous empirical $2$-NC and $6$-NC behavior.
\end{proposition}

\begin{remark}
The vanishing of these two character sums is an exact algebraic
consequence of the subgroup/coset pattern.  What remains conjectural is
that this distributional pattern holds for every lag in the limit
$N\to\infty$.
\end{remark}

\begin{remark}[Why this does not violate the 2-NC/4-NC dichotomy]
For $A_1$: at even $m$, $D_m \bmod 4 \in \{0, 1, 2, 3\}$
with approximately $P(0) \approx 0$, $P(2)\approx 0.5$,
$P(1)\approx 0.25$, $P(3)\approx 0.25$
(derived from the mod-$6$ data by reduction mod $4$).  This gives
$\delta^{\rm Re}_m \approx -0.5 \neq 0$, so $4$-NC fails at
those lags.  The sequence is empirically $2$-NC and $6$-NC but explicitly not
$4$-NC, confirming that the $2$-NC/$4$-NC dichotomy is not violated.
\end{remark}

\subsection{Exact rational values in the $k$-spectrum}

The exact values $G_{A_1, k, \infty}$ for non-NC $k$ can be computed
from the mod-$6$ distributions:

\begin{proposition}[Empirical spectrum of $A_1$]
\label{prop:exact_spectrum}
Assuming the observed law in \Cref{prop:26nc_obs}, one predicts
\begin{enumerate}
  \item $G_{A_1,3,N}\to 1$;
  \item $G_{A_1,4,N}\to \tfrac12$;
  \item $G_{A_1,5,N}\to \tfrac14$;
  \item $G_{A_1,8,N}\to 2^{-3/2}$.
\end{enumerate}
Numerically these limits are matched to at least three decimal digits
by $N=2^{16}$.
\end{proposition}

These values suggest that the non-NC spectrum is algebraic and governed
by character sums on finite residue groups.

\section{Conjectures and open problems}
\label{sec:open}
% ══════════════════════════════════════════════════════════════════════════════

\begin{conjecture}[2-NC / 4-NC dichotomy, restated]
\label{conj:dichotomy2}
For any finite $A \subset \{0,1\}^*$, the sequence $a_A$ cannot be
simultaneously $2$-NC and $4$-NC.  Equivalently, the $\{+1,-1\}$-valued
sequence $(-1)^{\occ(A,n)}$ is $2$-NC if and only if
$i^{\occ(A,n)}$ is \emph{not} $4$-NC.
\end{conjecture}

\begin{conjecture}[Small-modulus support]
\label{conj:k_values}
Within the family of binary pattern sequences, the only moduli that
admit $k$-NC examples up to $k\le 12$ are $k\in\{2,4,6\}$.
More strongly, we conjecture that no examples exist for odd $k$.
\end{conjecture}

\begin{conjecture}[Universal $O(1/N)$ decay]
For any $k$-NC binary pattern sequence $a_{A,k}$,
\[
  G_{A,k,N} = \frac{C_{A,k}}{N} + O\!\left(\frac{1}{N^2}\right)
\]
for a constant $C_{A,k} > 0$ depending only on $(A, k)$.
\end{conjecture}

\begin{conjecture}[Exact non-NC spectrum]
For any binary pattern set $A$ and any $k$ for which $a_{A,k}$ is
\emph{not} $k$-NC, the limiting value $G_{A,k,\infty}$ is a rational
linear combination of cosines of rational multiples of $\pi$, hence
an algebraic number.
\end{conjecture}

\noindent
\textbf{Key open problems.}
\begin{enumerate}
  \item Upgrade the observed parity taxonomy for the $k=4$ family to a
        theorem: characterize exactly which pattern sets force
        ALL-EVEN, ALTERNATING, MIXED, or MOME lag behavior.
  \item Prove or disprove the $2$-NC/$4$-NC dichotomy
        (\Cref{conj:dichotomy2}) for patterns of unbounded length.
  \item Extend the exact decision algorithm of Konieczny~\cite{Kon21}
        to $k$-NC for $k \geq 3$ by lifting the recursion to the
        cyclotomic ring $\Z[\omega_k]$.
  \item Determine whether $k = 8$ yields NC sequences for longer
        patterns (the search so far covers only length $\leq 4$).
  \item Classify all $k = 4$-NC binary pattern sequences,
        analogously to Konieczny's classification for $k = 2$.
\end{enumerate}
